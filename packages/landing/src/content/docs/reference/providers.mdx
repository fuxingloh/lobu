import skillsConfig from "@skills-config";
import { ProvidersRegistryTable } from "../../../components/ProvidersRegistryTable";

export const providerConfigEntries = (skillsConfig.skills || [])
  .filter((skill) => (skill.providers?.length || 0) > 0)
  .map((skill) => ({ id: skill.id, providers: skill.providers }));

Lobu model providers are config-driven and loaded from a single source of truth.

See also: [Skills](/getting-started/skills/) and [Capabilities](/getting-started/capabilities/).

## Source of truth

- Local file: `config/system-skills.json`
- GitHub source: [config/system-skills.json](https://github.com/lobu-ai/lobu/blob/main/config/system-skills.json)
